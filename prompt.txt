    def __init__(self, d_model, d_mid, dropout=0.1):
        super().__init__()

        self.up_proj = nn.Linear(d_mid, d_model)

    def forward(self, x):
        x = self.down_proj(x)
        x = self.activation(x)
        x = self.dropout(x)
        x = self.up_proj(x)
